# on_fleet_enter_orbit 
fleet_event = { 
    id = STH_hero_ship.3
    hide_window = yes 
    is_triggered_only = yes
	trigger = { exists = owner }
    immediate = { 
        save_event_target_as = fleet
		if = { 
            limit = { 
                exists = owner.capital_scope
				orbit = { is_same_value = owner.capital_scope }
            }
            random_owned_ship = {
                limit = {
                    fleet = { is_same_value = event_target:fleet }
                    has_ship_flag = hero_ship
                    has_ship_flag = nx-01
                    has_ship_flag = refit_planned
                }
                ship_event = { id = STH_hero_ship.4 }
            }
        } 
    } 
}

# Refit Ship
ship_event = {
    id = STH_hero_ship.5
    title = "STH_hero_ship.5.name"
    desc = "STH_hero_ship.5.desc"
    picture = sth_GFX_evt_nx01Refit3    
    is_triggered_only = yes
    trigger = {
        has_ship_flag = refit_in_progess
        has_ship_flag = hero_ship
        has_ship_flag = nx-01
    }
    immediate = {
        owner = { end_event_chain = "nx01_refit_event_chain" }
        fleet = {
            set_name = "Enterprise"
            set_event_locked = no
            spawn_hero_ship_ENT_FED = yes
			if = {
				limit = { exists = leader }
				leader = { 
					save_event_target_as = enterpriseLeader 
					unassign_leader = this
				}
			}
        }
        delete_ship = this
		fleet = {
			if = {
				limit = { exists = event_target:enterpriseLeader }
				fleet_event = { id = STH_hero_ship.8 scopes = { from = event_target:enterpriseLeader } days = 0 }
				fleet_event = { id = STH_hero_ship.8 scopes = { from = event_target:enterpriseLeader } days = 1 }
				fleet_event = { id = STH_hero_ship.8 scopes = { from = event_target:enterpriseLeader } days = 2 }
				fleet_event = { id = STH_hero_ship.8 scopes = { from = event_target:enterpriseLeader } days = 3 }
			}
		}
    }
    option = { name = sth_understood }
}
